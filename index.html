import random acords = {0: "C", 1: "Db", 2: "D", 3: "Eb", 4: "E", 5: "F", 6: "F#", 7: "G", 8: "Ab", 9: "A", 10: "Bb", 11: "B"} dades = ["0 5 7", "2 6 9", "4 0 7", "5 9 11"] def generar_progressio(durada, to): progressio_traduida = [] caracters = 0 while caracters < durada: acord_aleatori = random.choice(dades) acords_separats = acord_aleatori.split() if caracters + len(acords_separats) <= durada: for i in acords_separats: if 'm' in i: nota = (int(i[:-1]) + to) % 12 progressio_traduida.append(acords[nota] + 'm') else: i = int(i) nota_traduida = acords[(i + to) % 12] progressio_traduida.append(nota_traduida) caracters += len(acords_separats) return progressio_traduida def on_create_button_click(event): durada = int(Element("duration").value) tonalitat = Element("tonality").value to = list(acords.values()).index(tonalitat) progressio = generar_progressio(durada, to) Element("progression").element.innerHTML = f"Progressió: {' '.join(progressio)}" # Assignar l'event al botó Element("createProgression").element.addEventListener("click", on_create_button_click)
