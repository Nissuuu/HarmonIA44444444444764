HarmonIA: Genera la teva Progressió

Progressió:

import random from pyscript import Element acords = {0: "C", 1: "Db", 2: "D", 3: "Eb", 4: "E", 5: "F", 6: "F#", 7: "G", 8: "Ab", 9: "A", 10: "Bb", 11: "B"} dades = ["0 5 7", "2 6 9", "4 0 7", "5 9 11"] # Funció per generar la progressió def generar_progressio(durada, to): progressio_traduida = [] caracters = 0 while caracters < durada: acord_aleatori = random.choice(dades) acords_separats = acord_aleatori.split() if caracters + len(acords_separats) <= durada: for i in acords_separats: i = int(i) nota_traduida = acords[(i + to) % 12] progressio_traduida.append(nota_traduida) caracters += len(acords_separats) return progressio_traduida # Funció que es crida quan es fa clic al botó def on_create_button_click(event): durada_input = Element("duration").value tonalitat_input = Element("tonality").value # Convertir els valors d'entrada try: durada = int(durada_input) tonalitat = tonalitat_input to = list(acords.values()).index(tonalitat) # Generar progressió progressio = generar_progressio(durada, to) # Mostrar la progressió al paràgraf Element("progression").write(f"Progressió: {' '.join(progressio)}") except Exception as e: # Si hi ha algun error, el mostrarem per debugar Element("progression").write(f"Error: {str(e)}") # Assignar l'event al botó Element("createProgression").element.addEventListener("click", on_create_button_click)
